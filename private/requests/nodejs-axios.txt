var axios = require('axios');
var data = JSON.stringify({"hello":"world"});

var config = {
  method: 'post',
  url: '{{url}}',
  headers: { 
    'Content-Type': 'application/json'
  },
  data : data
};

axios(config)
.then(function (response) {
  console.log(JSON.stringify(response.data));
})
.catch(function (error) {
  console.log(error);
});
